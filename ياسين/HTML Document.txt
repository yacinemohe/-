    const resultsDiv = document.getElementById("results");
    const searchInput = document.getElementById("search");

    searchInput.addEventListener("input", () => {
      const query = searchInput.value.trim();
      resultsDiv.innerHTML = "";
      if (query === "") return;
      
      let currentCategory = "";
      
      accounts.forEach(account => {
        if (account.code.includes(query) || account.name.includes(query)) {
          if (account.category !== currentCategory) {
            currentCategory = account.category;
            const categoryDiv = document.createElement("div");
            categoryDiv.className = "category";
            categoryDiv.textContent = currentCategory;
            resultsDiv.appendChild(categoryDiv);
          }
          
          const div = document.createElement("div");
          div.className = "account";
          div.innerHTML = `<strong>${account.code}</strong>: ${account.name}`;
          resultsDiv.appendChild(div);
        }
      });
    });

    let currentQuestion;
    function startQuiz() {
      document.getElementById("quiz").style.display = "block";
      document.getElementById("result").innerText = "";
      document.getElementById("answer").value = "";
      currentQuestion = accounts[Math.floor(Math.random() * accounts.length)];
      document.getElementById("question").innerText = `ما اسم الحساب رقم: ${currentQuestion.code}؟`;
    }

    function checkAnswer() {
      const userAnswer = document.getElementById("answer").value.trim();
      const resultDiv = document.getElementById("result");
      if (userAnswer === currentQuestion.name) {
        resultDiv.innerText = "✔️ إجابة صحيحة!";
        resultDiv.style.color = "green";
      } else {
        resultDiv.innerText = `❌ خاطئ! الصحيح هو: ${currentQuestion.name}`;
        resultDiv.style.color = "red";
      }
    }
